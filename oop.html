Step 5 — Getter &amp; Setter
# Step 5: Getter and Setter — safe access to private data

class Animal:
    def __init__(self, name, age):
        self.__name = name
        self.__age  = age

    def get_name(self):           # getter — READ the value
        return self.__name

    def set_name(self, new_name): # setter — CHANGE with validation
        if len(new_name) > 0:
            self.__name = new_name
        else:
            print("Name cannot be empty!")

    def get_age(self):
        return self.__age

a = Animal("Luna", 3)
print(a.get_name())    # Luna

a.set_name("Mochi")
print(a.get_name())    # Mochi

a.set_name("")         # Name cannot be empty!
print(a.get_name())    # Mochi (unchanged)
💡 Naming convention: get_x() for getters, set_x() for setters. Setters can validate — rejecting empty strings, negative numbers, etc.

👨‍👧 Inheritance
Inheritance / super()
Inheritance lets a child class automatically get all attributes and methods from its parent.
class Dog(Animal) = Dog is a child (subclass) of Animal.
super().__init__(...) calls the parent's constructor first — so the parent's attributes are set up correctly.
🎨 Visual — Inheritance Tree: Child gets everything from Parent
↑ blue = inherited from Animal + green = Dog's own + purple = Cat's own
💡 super().__init__() runs Animal's constructor first — so Dog/Cat/Bird all get __name and __age automatically.
Step 6 — Inheritance
# Step 6: Inheritance — Dog gets everything Animal has

class Animal:
    def __init__(self, name, age):
        self.__name = name
        self.__age  = age

    def get_name(self):
        return self.__name

    def get_age(self):
        return self.__age

    def speak(self):
        return "..."

class Dog(Animal):                      # Dog INHERITS from Animal
    def __init__(self, name, age, breed):
        super().__init__(name, age)     # call Animal's __init__ first
        self.__breed = breed            # Dog's own extra attribute

    def get_breed(self):
        return self.__breed

rex = Dog("Rex", 2, "Labrador")

print(rex.get_name())    # Rex      — inherited from Animal
print(rex.get_age())     # 2        — inherited from Animal
print(rex.get_breed())   # Labrador — Dog's own method
print(rex.speak())       # ...      — inherited from Animal
💡 Think of it as a family tree. Dog is-a Animal. Everything Animal can do, Dog can do too — plus its own extra methods.

🎭 Polymorphism
Polymorphism / Override
Polymorphism = "many forms".
When a child class replaces a parent method with its own version, that's called overriding.
You can call the same method name on different objects and get different results — without needing to know which specific class you're dealing with.

# Step 7: Polymorphism — same method name, different behaviour

class Animal:

        self.__name = name
        self.__age  = age

    def get_name(self):
        return self.__name

    def speak(self):
        return "..."

class Dog(Animal):
    def speak(self):                        # OVERRIDE Animal.speak

class Cat(Animal):
    def speak(self):                        # OVERRIDE Animal.speak
        return f"{self.get_name()} says: Meow!"

    def speak(self):                        # OVERRIDE Animal.speak
        return f"{self.get_name()} says: Tweet!"

for a in animals:

# Luna says: Meow!
# Tweety says: Tweet!
💡 The for loop calls .speak() on every object. Python automatically picks the right version based on the actual class of each object.

# Step 8: Everything together — full OOP example

class Animal:                              # BASE CLASS (parent)
    def __init__(self, name, age):
        self.__name = name                 # ENCAPSULATION (private)
        self.__age  = age

    def get_name(self):                    # GETTER
        return self.__name

    def set_name(self, new_name):          # SETTER (with validation)
        if len(new_name) > 0:
            self.__name = new_name

    def get_age(self):
        return self.__age

    def describe(self):
        return f"Name: {self.__name}, Age: {self.__age}"

    def speak(self):
        return "..."


class Dog(Animal):                         # INHERITANCE (child)
    def __init__(self, name, age, breed):
        super().__init__(name, age)        # call parent __init__
        self.__breed = breed

    def get_breed(self):
        return self.__breed

    def speak(self):                       # POLYMORPHISM (override)
        return f"{self.get_name()} says: Woof!"

    def describe(self):
        return f"{super().describe()}, Breed: {self.__breed}"


class Cat(Animal):
    def __init__(self, name, age, indoor):
        super().__init__(name, age)
        self.__indoor = indoor

    def is_indoor(self):
        return self.__indoor

    def speak(self):
        return f"{self.get_name()} says: Meow!"


# Create objects
rex  = Dog("Rex", 2, "Labrador")
luna = Cat("Luna", 3, True)

print(rex.describe())     # Name: Rex, Age: 2, Breed: Labrador
print(luna.describe())    # Name: Luna, Age: 3

print(rex.speak())        # Rex says: Woof!
print(luna.speak())       # Luna says: Meow!

rex.set_name("Max")
print(rex.get_name())     # Max  (changed via setter)

class BankAccount:
    def __init__(self, owner, balance):
        self.__owner   = owner
        self.__balance = balance

    def deposit(self, amount):
        self.__balance += amount

    def withdraw(self, amount):
        if amount <= self.__balance:
            self.__balance -= amount
        else:
            print("Insufficient funds")

    def get_balance(self):
        return self.__balance

    def get_owner(self):
        return self.__owner


class SavingsAccount(BankAccount):
    def add_interest(self, rate):
        interest = self.get_balance() * rate
        self.deposit(interest)

🆓 L3 — Free Style: Shape Hierarchy

Design a Shape class hierarchy.
Base: Shape — defines area() and perimeter().
Subclasses: Circle, Rectangle, Triangle — each overrides both methods.

💡 Hint: import math for math.pi and math.sqrt(). Use raise NotImplementedError in the base class.
answer.py
import math

class Shape:
    def area(self):
        raise NotImplementedError("Subclass must implement area()")

    def perimeter(self):
        raise NotImplementedError("Subclass must implement perimeter()")

    def describe(self):
        return (f"{type(self).__name__}: "
                f"area={self.area()}, "
                f"perimeter={self.perimeter()}")


class Circle(Shape):
    def __init__(self, radius):
        self.__radius = radius

    def area(self):
        return round(math.pi * self.__radius ** 2, 2)

    def perimeter(self):
        return round(2 * math.pi * self.__radius, 2)


class Rectangle(Shape):
    def __init__(self, width, height):
        self.__width  = width
        self.__height = height

    def area(self):
        return self.__width * self.__height

    def perimeter(self):
        return 2 * (self.__width + self.__height)


class Triangle(Shape):
    def __init__(self, a, b, c):
        self.__a = a
        self.__b = b
        self.__c = c

    def area(self):
        s = (self.__a + self.__b + self.__c) / 2
        return round(math.sqrt(s * (s-self.__a) * (s-self.__b) * (s-self.__c)), 2)

    def perimeter(self):
        return self.__a + self.__b + self.__c


shapes = [Circle(5), Rectangle(4, 6), Triangle(3, 4, 5)]
for s in shapes:
    print(s.describe())
# Circle: area=78.54, perimeter=31.42
# Rectangle: area=24, perimeter=20
# Triangle: area=6.0, perimeter=12

            self.__name = new_name

    def get_info(self):
        return f"Name: {self.__name}, Age: {self.__age}"


class Student(Person):               # inheritance
    def __init__(self, name, age, grade, subjects):
        super().__init__(name, age)
        self.__grade    = grade
        self.__subjects = subjects

    def get_grade(self):
        return self.__grade

    def get_info(self):              # polymorphism (override)
        base = super().get_info()
        return (f"[Student] {base}, "
                f"Grade: {self.__grade}, "
                f"Subjects: {', '.join(self.__subjects)}")


class Teacher(Person):               # inheritance
    def __init__(self, name, age, subject, salary):
        super().__init__(name, age)
        self.__subject = subject
        self.__salary  = salary

    def get_info(self):              # polymorphism (override)
        base = super().get_info()
        return (f"[Teacher] {base}, "
                f"Teaches: {self.__subject}, "
                f"Salary: ${self.__salary:,}")


people = [
    Student("Alice", 15, "10A", ["Math", "CS", "Physics"]),
    Student("Bob",   14, "9B",  ["English", "Art"]),
    Student("Carol", 16, "11A", ["CS", "Biology"]),
    Teacher("Mr. Smith", 40, "Computer Science", 45000),
    Teacher("Ms. Lee",   35, "Mathematics",      42000),
]

for p in people:
    print(p.get_info())
# Encapsulation  → __name, __age, __grade, __salary are private
# Inheritance    → Student/Teacher inherit from Person
# Polymorphism   → get_info() works differently per class
