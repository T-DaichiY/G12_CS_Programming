Code 1 — Student Report Card
①
🏛️ OOP Concepts Used
Demonstrates the core pillars of OOP: Encapsulation (private __grades dict), Abstraction (user calls get_report() without knowing internals). A clean real-world model — data + behaviour in one class.
Encapsulation
Private data
self.__grades can't be accessed directly from outside.
Abstraction
Hidden logic
Caller uses get_report() — doesn't need to know how grades are stored.
②
🐍 Student Report Card
student_report.py
class Student: def __init__(self, name, student_id): self.__name = name self.__id = student_id self.__grades = {} def add_grade(self, subject, score): if 0 <= score <= 100: self.__grades[subject] = score else: print("Score must be 0–100") def get_average(self): if not self.__grades: return 0 return sum(self.__grades.values()) / len(self.__grades) def get_report(self): print(f"\n📋 Report Card — {self.__name} (ID: {self.__id})") print("-" * 35) for subject, score in self.__grades.items(): grade = "A" if score >= 80 else "B" if score >= 70 else "C" if score >= 60 else "F" print(f" {subject:<15} {score:>3} [{grade}]") print(f" Average {self.get_average():>5.1f}") s1 = Student("Daichi", "S001") s1.add_grade("Computer Science", 91) s1.add_grade("Mathematics", 85) s1.add_grade("Physics", 78) s1.get_report()

Code 2 — Number Guessing Game 🎮
①
🎮 Why OOP for Games?
Shows how OOP is used to model game state. The NumberGame class encapsulates the secret number so it can't be read directly. The Player class tracks stats independently — demonstrating separation of concerns.
State Hiding
self.__secret
The secret number is private — users must call guess(), not read it directly.
Separation
2 classes
Player tracks stats. NumberGame handles logic. Each has one job.
②
🐍 Number Guessing Game
number_game.py
import random class Player: def __init__(self, name): self.name = name self.attempts = 0 self.wins = 0 def record_win(self, attempts): self.wins += 1 self.attempts += attempts def get_stats(self): avg = self.attempts / self.wins if self.wins > 0 else 0 return f"{self.name}: {self.wins} wins, avg {avg:.1f} attempts" class NumberGame: def __init__(self, player, low=1, high=100): self.__player = player self.__low = low self.__high = high self.__secret = random.randint(low, high) self.__attempts = 0 self.__max_attempts = 7 def guess(self, number): self.__attempts += 1 if number < self.__low or number > self.__high: return f"Out of range! Guess between {self.__low} and {self.__high}." if number < self.__secret: return f"Too low! ({self.__max_attempts - self.__attempts} guesses left)" elif number > self.__secret: return f"Too high! ({self.__max_attempts - self.__attempts} guesses left)" else: self.__player.record_win(self.__attempts) return f"Correct! You got it in {self.__attempts} attempts!" def is_over(self): return self.__attempts >= self.__max_attempts def reveal(self): return f"The number was {self.__secret}." # --- Play --- player = Player("Daichi") game = NumberGame(player) for g in [50, 75, 60, 55, 52, 54, 53]: result = game.guess(g) print(f"Guess {g}: {result}") if "Correct" in result or game.is_over(): break if game.is_over() and "Correct" not in result: print(game.reveal()) print(player.get_stats())

Code 3 — RPG Battle ⚔️
①
⚔️ Inheritance in Action
The flagship OOP example — shows all four pillars together. Hero and Monster inherit from Character, each overriding or extending behaviour. take_damage() is called polymorphically on any Character.
Inheritance
Hero, Monster
Both inherit attack(), take_damage(), status() from Character.
Polymorphism
same method name
target.take_damage() works on both Hero and Monster — same call, different objects.
②
🐍 RPG Battle System
rpg_battle.py
import random class Character: def __init__(self, name, hp, attack, defense): self.name = name self.__hp = hp self.__max_hp = hp self.attack_power = attack self.defense = defense def take_damage(self, amount): damage = max(0, amount - self.defense) self.__hp = max(0, self.__hp - damage) return damage def attack(self, target): dmg = random.randint(self.attack_power - 3, self.attack_power + 3) return target.take_damage(dmg) def is_alive(self): return self.__hp > 0 def get_hp(self): return self.__hp def status(self): bar = "█" * (self.__hp * 10 // self.__max_hp) return f"{self.name:<12} HP: [{bar:<10}] {self.__hp}/{self.__max_hp}" class Hero(Character): def __init__(self, name): super().__init__(name, hp=100, attack=15, defense=5) self.__potions = 2 def use_potion(self): if self.__potions > 0: heal = random.randint(20, 35) self.__potions -= 1 self.__dict__['_Character__hp'] = min( self.__dict__['_Character__max_hp'], self.__dict__['_Character__hp'] + heal ) return f"{self.name} used a potion! (+{heal} HP) [{self.__potions} left]" return "No potions left!" class Monster(Character): def __init__(self, name, level): super().__init__(name, hp=60 + level*10, attack=10 + level*2, defense=level) self.level = level def special_attack(self, target): return target.take_damage(self.attack_power * 2) # --- Battle --- hero = Hero("Daichi") monster = Monster("Dragon", level=3) print("BATTLE START!") round_num = 1 while hero.is_alive() and monster.is_alive(): print(f"--- Round {round_num} ---") dmg = hero.attack(monster) print(f" {hero.name} hits {monster.name} for {dmg} damage!") if hero.get_hp() < 40: print(" ", hero.use_potion()) if round_num % 3 == 0: dmg = monster.special_attack(hero) print(f" {monster.name} SPECIAL ATTACKS for {dmg} damage!") else: dmg = monster.attack(hero) print(f" {monster.name} hits {hero.name} for {dmg} damage!") print(f" {hero.status()}") print(f" {monster.status()}") round_num += 1 print("Hero wins!" if hero.is_alive() else "Monster wins...")

+{healed} HP [{self.__potions} left]" return "No potions!" def critical_strike(self, target): dmg = target.take_damage(self.attack_power * 2) return f"CRITICAL! {dmg} damage!" class Monster(Character): def __init__(self, name, level): super().__init__(name, hp=60+level*10, attack=10+level*2, defense=level) self.level = level self.__rage = 0 def take_damage(self, amount): dmg = super().take_damage(amount) self.__rage = min(10, self.__rage + 1) # rage builds return dmg def rage_attack(self, target): bonus = self.__rage * 2 dmg = target.take_damage(self.attack_power + bonus) self.__rage = 0 return f"RAGE ATTACK (+{bonus})!
