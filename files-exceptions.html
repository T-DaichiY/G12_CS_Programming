🆓 L3 — Free Style

Write a grade book system: add_student(filename, name, score), get_top_student(filename), and get_class_average(filename). All functions must include exception handling.

answer.py
# Grade Book System — file I/O with exception handling

def add_student(filename, name, score):
    try:
        with open(filename, 'a') as f:
            f.write(f"{name},{score}\n")
        print(f"Added: {name} ({score})")
    except IOError as e:
        print(f"Error writing to file: {e}")

def get_top_student(filename):
    try:
        with open(filename, 'r') as f:
            lines = [line.strip().split(',') for line in f if line.strip()]
        if not lines:
            return None, 0
        top = max(lines, key=lambda x: int(x[1]))
        return top[0], int(top[1])
    except FileNotFoundError:
        print("Grade book not found.")
        return None, 0
    except ValueError:
        print("Invalid data in file.")
        return None, 0

def get_class_average(filename):
    try:
        with open(filename, 'r') as f:
            lines = [line.strip().split(',') for line in f if line.strip()]
        if not lines:
            return 0.0
        scores = [int(row[1]) for row in lines]
        return sum(scores) / len(scores)
    except FileNotFoundError:
        print("Grade book not found.")
        return 0.0
    except ValueError:
        print("Invalid score in file.")
        return 0.0

# ── Test ──
import os
filename = "gradebook.txt"
if os.path.exists(filename):
    os.remove(filename)

add_student(filename, "Alice", 92)
add_student(filename, "Bob", 78)
add_student(filename, "Carol", 88)
add_student(filename, "Dave", 95)

name, score = get_top_student(filename)
print(f"Top student: {name} ({score})")
avg = get_class_average(filename)
print(f"Class average: {avg:.1f}")

# Output:
# Added: Alice (92)
# Added: Bob (78)
# Added: Carol (88)
# Added: Dave (95)
# Top student: Dave (95)
# Class average: 88.2

answer.py
# School Canteen Inventory System (CSV format)

import os

FILENAME = "inventory.txt"

def load_inventory(filename):
    inventory = {}
    try:
        with open(filename, 'r') as f:
            for line in f:
                line = line.strip()
                if line:
                    name, qty = line.split(',')
                    inventory[name] = int(qty)
    except FileNotFoundError:
        pass  # first run — empty inventory
    except ValueError as e:
        print(f"Data error in file: {e}")
    return inventory

def save_inventory(filename, inventory):
    try:
        with open(filename, 'w') as f:
            for item, qty in sorted(inventory.items()):
                f.write(f"{item},{qty}\n")
    except IOError as e:
        print(f"Save failed: {e}")

def add_item(filename, item_name, quantity):
    try:
        inv = load_inventory(filename)
        if item_name in inv:
            print(f"'{item_name}' already exists. Use update_quantity().")
        else:
            inv[item_name] = quantity
            save_inventory(filename, inv)
            print(f"Added: {item_name} x{quantity}")
    except Exception as e:
        print(f"Error adding item: {e}")

def update_quantity(filename, item_name, new_qty):
    try:
        inv = load_inventory(filename)
        if item_name not in inv:
            raise KeyError(f"'{item_name}' not found in inventory")
        inv[item_name] = new_qty
        save_inventory(filename, inv)
        print(f"Updated: {item_name} → qty {new_qty}")
    except KeyError as e:
        print(f"Update failed: {e}")
    except ValueError:
        print("Invalid quantity — must be an integer.")

def remove_item(filename, item_name):
    try:
        inv = load_inventory(filename)
        if item_name not in inv:
            raise KeyError(f"'{item_name}' not found")
        del inv[item_name]
        save_inventory(filename, inv)
        print(f"Removed: {item_name}")
    except KeyError as e:
        print(f"Remove failed: {e}")

def display_inventory(filename):
    try:
        inv = load_inventory(filename)
        if not inv:
            print("Inventory is empty.")
            return
        print("\n--- Canteen Inventory ---")
        for item, qty in sorted(inv.items()):
            print(f"  {item:<20} qty: {qty}")
        print("-------------------------")
    except Exception as e:
        print(f"Display error: {e}")

# ── Test ──
if os.path.exists(FILENAME):
    os.remove(FILENAME)

add_item(FILENAME, "Water Bottle", 50)
add_item(FILENAME, "Sandwich", 20)
add_item(FILENAME, "Chips", 100)
display_inventory(FILENAME)

update_quantity(FILENAME, "Water Bottle", 45)
remove_item(FILENAME, "Chips")
remove_item(FILENAME, "Pizza")    # KeyError case
display_inventory(FILENAME)

# Output:
# Added: Water Bottle x50
# Added: Sandwich x20
# Added: Chips x100
# --- Canteen Inventory ---
#   Chips                qty: 100
#   Sandwich             qty: 20
#   Water Bottle         qty: 50
# -------------------------
# Updated: Water Bottle → qty 45
# Removed: Chips
# Remove failed: 'Pizza' not found
# --- Canteen Inventory ---
#   Sandwich             qty: 20
#   Water Bottle         qty: 45
# -------------------------
📝 Key Terms
"r" / "w" / "a"
file mode
Read / Write (overwrite) / Append (add to end)
with open()
context manager
Automatically closes the file when done
try / except
exception handling
Catch and handle errors gracefully
finally
always runs
Executes whether or not an exception occurred
FileNotFoundError
common exception
Raised when trying to open a non-existent file
