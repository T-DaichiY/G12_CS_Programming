🆓 L3 — Free Style

Build a complete BinarySearchTree class from scratch with insert() and find() methods. Then insert: 50, 30, 70, 20, 40, 60, 80. Find 40 and 99 and print results.

💡 Hint: For insert — if value < node go left, if value > node go right. For find — same logic.
answer.py
# Binary Search Tree — insert() and find()

class Node:
    def __init__(self, value):
        self.value = value
        self.left = None
        self.right = None

class BinarySearchTree:
    def __init__(self):
        self.root = None

    def insert(self, value):
        if self.root is None:
            self.root = Node(value)
        else:
            self._insert(self.root, value)

    def _insert(self, node, value):
        if value < node.value:
            if node.left is None:
                node.left = Node(value)
            else:
                self._insert(node.left, value)
        elif value > node.value:
            if node.right is None:
                node.right = Node(value)
            else:
                self._insert(node.right, value)
        # equal values: ignore (no duplicates)

    def find(self, value):
        return self._find(self.root, value)

    def _find(self, node, value):
        if node is None:
            return False
        if value == node.value:
            return True
        elif value < node.value:
            return self._find(node.left, value)
        else:
            return self._find(node.right, value)

# Insert: 50, 30, 70, 20, 40, 60, 80
bst = BinarySearchTree()
for val in [50, 30, 70, 20, 40, 60, 80]:
    bst.insert(val)

# Tree structure:
#          50
#         /  \
#       30    70
#      / \   / \
#    20  40 60  80

print(bst.find(40))   # True
print(bst.find(99))   # False
print(bst.find(60))   # True
🎯 L4 — Scenario
Situation
A hospital uses a queue to manage patient appointments and a stack to manage the history of operations performed (so the last operation can be undone). Implement both, then simulate: 5 patients arrive, 2 are seen, 3 operations are performed, 1 is undone.
  • Implement Queue (enqueue/dequeue) and Stack (push/pop) classes
  • Simulate arrival of 5 patients into the queue
  • Dequeue 2 patients (they are being seen)
  • Push 3 operations onto the stack
  • Pop 1 to "undo" and print what was undone
  • Print final state of queue and stack
answer.py
# Hospital Queue + Operation Stack simulation

class Queue:
    def __init__(self):
        self.__items = []

    def enqueue(self, item):
        self.__items.append(item)

    def dequeue(self):
        if self.is_empty():
            raise IndexError("Queue is empty — no patients waiting")
        return self.__items.pop(0)

    def is_empty(self):
        return len(self.__items) == 0

    def size(self):
        return len(self.__items)

    def display(self):
        print(f"  Waiting queue ({self.size()}): {self.__items}")


class Stack:
    def __init__(self):
        self.__items = []

    def push(self, item):
        self.__items.append(item)

    def pop(self):
        if self.is_empty():
            raise IndexError("Stack is empty — nothing to undo")
        return self.__items.pop()

    def is_empty(self):
        return len(self.__items) == 0

    def peek(self):
        return self.__items[-1] if not self.is_empty() else None


# ── Simulation ──
queue   = Queue()     # patients waiting
history = Stack()     # completed operations (undo-able)

print("=== 5 patients arrive ===")
for p in ["Alice", "Bob", "Carol", "Dave", "Eve"]:
    queue.enqueue(p)
    print(f"  Arrived: {p}")
queue.display()

print("\n=== 2 patients are seen ===")
p1 = queue.dequeue()
print(f"  Now seeing: {p1}")
p2 = queue.dequeue()
print(f"  Now seeing: {p2}")
queue.display()

print("\n=== 3 operations recorded ===")
ops = [
    f"Blood test for {p1}",
    f"X-ray for {p2}",
    "Prescription for Carol",
]
for op in ops:
    history.push(op)
    print(f"  Recorded: {op}")
print(f"  Last op: {history.peek()}")

print("\n=== Undo last operation ===")
undone = history.pop()
print(f"  Undone: {undone}")
